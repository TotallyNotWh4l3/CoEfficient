Shorten Directory...

function prompt {
    $parent = split-path (split-path $pwd -parent) -leaf
    $current = split-path $pwd -leaf

    if ($parent) {
        "PS ...\$parent\$current> "
    } else {
        "PS $current> "
    }
}
